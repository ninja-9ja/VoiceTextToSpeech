# Создание образа из докерфайла находясь в папке с докерфайлом
docker build -t qwen-tts .

# Запуск докера на основе образа с учетом портов
docker run --rm -p 8000:8000 --name qwen-tts qwen-tts

# Просмотр логов запущенного докера
docker logs -f --tail 50 20a6380b341e

# Вход в докер контейнер
docker exec -ti 3e877b8b987c bash

# Удалить зависший контейнер
docker rm -f e6e787d7646c

Проверка, что сервер жив:
curl http://127.0.0.1:8000/health
Ожидаемо:
{"status":"ok"}

Основной тест TTS
curl -v -X POST http://127.0.0.1:8000/tts \
  -H "Content-Type: application/json" \
  -d '{"text":"Hello! This is a test."}' \
  --output output.wav

Тест клонирования русского голоса:
curl -v -X POST http://127.0.0.1:8000/tts-clone \
  -F 'text=Привет! Это уже новый текст, но сказанный клонированным голосом на русском языке.' \
  -F 'ref_text=Нет киса - это мой пирог. Брысь! Плохая киса! Плохая-плохая, это моя запеканка! Брысь от сюда - пошла вон. Мам киса прям как виброжоп.' \
  -F 'ref_audio=@Eric_Cartman.wav;type=audio/wav' \
  --output output.wav
